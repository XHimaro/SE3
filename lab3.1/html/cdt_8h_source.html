219/* special search function for tree structure only */
220#define _DTMTCH(dt,key,action) \
221 do { Dtlink_t* _e; void *_o, *_k, *_key; Dtdisc_t* _dc; \
222 int _ky, _sz, _lk, _cmp; Dtcompar_f _cmpf; \
223 _dc = (dt)->disc; _DTDSC(_dc, _ky, _sz, _lk, _cmpf); \
224 _key = (key); \
225 for(_e = (dt)->data->here; _e; _e = _cmp < 0 ? _e->hl._left : _e->right) \
226 { _o = _DTOBJ(_e, _lk); _k = _DTKEY(_o, _ky, _sz); \
227 if((_cmp = _DTCMP((dt), _key, _k, _dc, _cmpf, _sz)) == 0) \
228 break; \
229 } \
230 action (_e ? _o : (void*)0); \
231 } while(0)
232
233#define _DTSRCH(dt,obj,action) \
234 do { Dtlink_t* _e; void *_o, *_k, *_key; Dtdisc_t* _dc; \
235 int _ky, _sz, _lk, _cmp; Dtcompar_f _cmpf; \
236 _dc = (dt)->disc; _DTDSC(_dc, _ky, _sz, _lk, _cmpf); \
237 _key = _DTKEY(obj, _ky, _sz); \
238 for(_e = (dt)->data->here; _e; _e = _cmp < 0 ? _e->hl._left : _e->right) \
239 { _o = _DTOBJ(_e, _lk); _k = _DTKEY(_o, _ky, _sz); \
240 if((_cmp = _DTCMP((dt), _key, _k, _dc, _cmpf, _sz)) == 0) \
241 break; \
242 } \
243 action (_e ? _o : (void*)0); \
244 } while(0)
